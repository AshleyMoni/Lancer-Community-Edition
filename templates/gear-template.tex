% Weapon Tags
\usepackage{pgfkeys}

\providecommand{\fluff}[1]{{\itshape #1}}

\pgfkeys {
  % we define a family, and enter it
  /gear/.is family, /gear,
  % we define some defaults
  default/.style =
  {
    name = undefined,
    fluff = undefined,
    template = undefined,
    rules = {}
  },
  name/.store in = \gearName,
  fluff/.store in = \gearFluff,
  template/.store in = \gearTemplate,
  rules/.store in = \gearRules
}
\newcommand\gearBox[1][] {
  \pgfkeys{/gear, default, #1}
  \textbf{\gearName} \\
  Fluff: \fluff{\gearFluff} \\
  Template: \gearTemplate \\
  Rules: \gearRules
}
